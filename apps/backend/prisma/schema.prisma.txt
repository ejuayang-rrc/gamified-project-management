generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  // Change to vercel for PROD deployment
  // Use localhost:..... for dev. testing
  url      = env("DATABASE_URL")
}

model Task {
    id          Int     @id @default(autoincrement())
    taskId      Int     @unique
    assignedTo  User    @relation(fields: [assignedId], references:[id])
    assignedId  Int
    AssignedOn  String
    dueDate     String
    difficulty  String
    status      Boolean
}

model User {
    id          Int      @id @default(autoincrement())
    username    String   
    email       String   @unique
    group       Group    @relation(fields: [groupId], references: [id])
    groupId     Int      
    tasks Task[]
}

model Group {
    id             Int             @id @default(autoincrement())
    groupId        Int             @unique
    name           String   
    numberOfUsers  Int
    organization   Organization    @relation(fields: [organizationId], references:[id])      
    organizationId Int
}

model Organization {
    id                  Int    @id @default(autoincrement())
    organizationId      Int
    groups              Group[]
    name                String
}
